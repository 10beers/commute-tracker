<body class="bg-gray-100 p-8">

    <div class="max-w-2xl mx-auto bg-white p-6 rounded-xl shadow-md">

        <h1 class="text-2xl font-bold mb-2">Commute Tracker</h1>

        <p class="text-sm text-gray-500 mb-6">Checking every 30 mins: 05:00-08:30 & 15:30-18:30</p>

        

        <div class="space-y-4">

            <div>

                <label class="block text-sm font-medium">Google Maps API Key</label>

                <input type="password" id="apiKey" class="w-full p-2 border rounded" placeholder="Paste Key Here">

            </div>

            <div>

                <label class="block text-sm font-medium">Destination</label>

                <input type="text" id="destination" class="w-full p-2 border rounded" value="1600 Amphitheatre Pkwy, Mountain View, CA">

            </div>

            <div>

                <label class="block text-sm font-medium">Origins (one per line)</label>

                <textarea id="origins" class="w-full p-2 border rounded" rows="3">San Jose, CA&#10;San Francisco, CA&#10;Oakland, CA</textarea>

            </div>

            <button onclick="toggleTracking()" id="trackBtn" class="bg-blue-600 text-white px-6 py-2 rounded-full font-bold">Start Tracking</button>

            

            <div class="grid grid-cols-2 gap-4 pt-4 border-t">

                <div>

                    <p class="text-xs text-gray-400 uppercase">Status</p>

                    <p id="statusText" class="font-bold text-gray-600">Idle</p>

                </div>

                <div>

                    <p class="text-xs text-gray-400 uppercase">Last Update</p>

                    <p id="lastUpdate" class="font-bold">--:--</p>

                </div>

            </div>


            <div id="results" class="mt-4 space-y-2"></div>

        </div>

    </div>


    <script>

        let isTracking = false;

        let timer = null;


        function isWithinWindow() {

            const now = new Date();

            const hour = now.getHours() + now.getMinutes() / 60;

            return (hour >= 5 && hour <= 8.5) || (hour >= 15.5 && hour <= 18.5);

        }


        async function fetchData() {

            if (!isWithinWindow()) {

                document.getElementById('statusText').innerText = "Outside Window - Sleeping";

                return;

            }


            const key = document.getElementById('apiKey').value;

            const dest = document.getElementById('destination').value;

            const origins = document.getElementById('origins').value.split('\n').filter(o => o.trim() !== "");

            

            document.getElementById('statusText').innerText = "Updating...";

            

            try {

                const response = await fetch(`https://maps.googleapis.com/maps/api/distancematrix/json?origins=${encodeURIComponent(origins.join('|'))}&destinations=${encodeURIComponent(dest)}&key=${key}`);

                const data = await response.json();

                

                let html = "";

                data.rows.forEach((row, i) => {

                    const time = row.elements[0].duration.text;

                    html += `<div class="flex justify-between border-b py-1"><span>${origins[i]}</span><span class="font-mono">${time}</span></div>`;

                });

                

                document.getElementById('results').innerHTML = html;

                document.getElementById('lastUpdate').innerText = new Date().toLocaleTimeString();

                document.getElementById('statusText').innerText = "Active - Monitoring";

            } catch (e) {

                document.getElementById('statusText').innerText = "Error - Check API Key";

            }

        }


        function toggleTracking() {

            isTracking = !isTracking;

            const btn = document.getElementById('trackBtn');

            

            if (isTracking) {

                btn.innerText = "Stop Tracking";

                btn.className = "bg-red-600 text-white px-6 py-2 rounded-full font-bold";

                fetchData();

                timer = setInterval(fetchData, 30 * 60 * 1000); // 30 mins

            } else {

                btn.innerText = "Start Tracking";

                btn.className = "bg-blue-600 text-white px-6 py-2 rounded-full font-bold";

                clearInterval(timer);

                document.getElementById('statusText').innerText = "Idle";

            }

        }

    </script>

</body>
